\begin{frame}{Goal}
\begin{itemize}
  \item Integrate GUAVA-style end-to-end ATAC-seq analysis capabilities into the existing modular Snakemake pipeline.
  \item Prioritize missing capabilities: differential analysis, functional enrichment, and presentation-ready outputs.
  \item Keep implementation reproducible, configurable, and suitable for batch processing.
\end{itemize}
\end{frame}

\begin{frame}{Current Pipeline Status}
\textbf{Already implemented in this pipeline}
\begin{itemize}
  \item Preprocessing and QC: fastp, FastQC, MultiQC.
  \item Alignment and filtering: Bowtie2, mito filtering, deduplication, MAPQ filtering, Tn5 shifting.
  \item Peak workflow: MACS2 peak calling, blacklist filtering, peak annotation, motif analysis.
  \item Signal and QC summaries: bigWig generation, correlation, FRiP, TSS enrichment, preseq, qualimap.
\end{itemize}

\textbf{Main gap vs GUAVA}
\begin{itemize}
  \item No integrated differential ATAC module between conditions with replicates.
  \item No integrated GO/KEGG enrichment on differential peaks.
  \item No consolidated presentation bundle for differential interpretation.
\end{itemize}
\end{frame}

\begin{frame}{GUAVA-Inspired Target Architecture}
\begin{itemize}
  \item \textbf{Part 1 (already mostly present):} sample-level processing and peak calling.
  \item \textbf{Part 2 (to add):} condition-level differential peak analysis.
  \item Add rule chain:
  \begin{itemize}
    \item Consensus peak generation.
    \item Read counting on consensus peaks.
    \item DESeq2 differential testing.
    \item Peak-to-gene annotation and GO/KEGG enrichment.
    \item Volcano/PCA/MA plots and summary tables.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Phase-wise Implementation}
\textbf{Phase 1: Metadata and design matrix}
\begin{itemize}
  \item Create sample metadata table with condition and replicate columns.
  \item Validate at least 2 replicates per condition for robust differential calls.
\end{itemize}

\textbf{Phase 2: Differential peak discovery}
\begin{itemize}
  \item Merge replicate peaks into consensus set.
  \item Count reads per consensus peak for each sample.
  \item Run DESeq2 with configured contrast and cutoff.
\end{itemize}

\textbf{Phase 3: Biological interpretation}
\begin{itemize}
  \item Annotate differential peaks to nearest genes.
  \item Run GO/KEGG enrichment and export tables + plots.
\end{itemize}

\textbf{Phase 4: Reporting and presentation}
\begin{itemize}
  \item Generate volcano plot, PCA plot, MA plot, gained-open/closed summary.
  \item Integrate summaries into MultiQC companion report.
\end{itemize}
\end{frame}

\begin{frame}{Deliverables For Presentation}
\begin{itemize}
  \item Differential peak table: peak ID, log2FC, pvalue, padj, status.
  \item Volcano plot and MA plot for contrast.
  \item PCA of samples using peak count matrix.
  \item GO/KEGG enriched terms table and bar/dot plot.
  \item Annotation summary (promoter/intron/exon/intergenic distribution).
  \item IGV-ready bundle: normalized bigWig + differential peak BED files.
\end{itemize}
\end{frame}

\begin{frame}{Validation Checklist}
\begin{itemize}
  \item Technical QC checks pass before differential analysis:
  \begin{itemize}
    \item FRiP acceptable range.
    \item TSS enrichment acceptable range.
    \item Duplicate and mito rates within thresholds.
  \end{itemize}
  \item Replicates cluster by biological condition in PCA.
  \item Differential peak direction consistent with IGV tracks.
  \item Enrichment terms biologically plausible for study context.
\end{itemize}
\end{frame}
